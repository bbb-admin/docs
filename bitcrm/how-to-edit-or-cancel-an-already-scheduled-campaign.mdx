import {Step, StepList, Alert} from "@mintlify/components";

## Step 1: Go to bitCRM Campaign Dashboard

Navigate to your [bitCRM Campaign dashboard](https://app.bitbybit.studio).

<img
  src="/images/bitcrm-campaign-dashboard.png"
  alt="bitCRM Campaign Dashboard"
  className="rounded-lg"
/>

## Step 2: Click on the Campaign to Make Edits

Click on the campaign you wish to edit.

<img
  src="/images/click-campaign-to-edit.png"
  alt="Click Campaign to Edit"
  className="rounded-lg"
/>

## Step 3: Click the Checkbox Next to the Campaign, Then Click 'Delete' to Remove It

To cancel the campaign, click the checkbox next to the campaign, then click "Delete" to remove it.

<img
  src="/images/select-campaign-and-delete.png"
  alt="Select Campaign and Delete"
  className="rounded-lg"
/>
