import {Step, StepList, Alert} from "@mintlify/components";

## Step 1: Go to WhatsApp API Profile Settings

1. Log in to your bitbybit account.
2. Navigate to the **WhatsApp API Profile** settings.

<img
  src="/images/whatsapp-api-profile-settings.png"
  alt="WhatsApp API Profile Settings"
  className="rounded-lg"
/>

## Step 2: Click 'Upload Image' on Profile Photo

1. In the Profile Photo section, click on **'Upload image'**.

<img
  src="/images/upload-image-button.png"
  alt="Upload Image Button"
  className="rounded-lg"
/>

## Step 3: Click 'Save Changes'

1. After selecting your image, click **'Save changes'** to update your profile picture.

<img
  src="/images/save-changes-button.png"
  alt="Save Changes Button"
  className="rounded-lg"
/>
